/* TIMELINE FIXED */
function updateTimeline(){

  let start = new Date("2024-08-14");
  let now = new Date();

  let diff = now - start;

  let days = Math.floor(diff / (1000*60*60*24));

  let years = Math.floor(days / 365);
  let months = Math.floor((days % 365) / 30);

  document.getElementById("timeline").innerHTML =
  `💞 Together Since 14 Aug 2024<br>
   📅 Total Days: ${days}<br>
   📆 ${years} Year ${months} Months<br>
   💖 Still Strong ❤️`;
}

updateTimeline();
